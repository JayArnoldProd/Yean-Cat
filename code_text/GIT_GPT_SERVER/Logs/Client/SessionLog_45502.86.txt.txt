/// @function scr_set_target(target)
/// @param {string/real} target - Instance ID or reference name of the target
function scr_set_target(target) {
    if (is_string(target)) {
        var obj = get_object_reference(target);
        if (obj != noone) {
            global.currentTarget = obj;
            return "Target set to " + target;
        } else {
            return "Invalid target reference: " + target;
        }
    } else if (instance_exists(target)) {
        global.currentTarget = target;
        return "Target set to instance " + string(target);
    } else {
        return "Invalid target instance ID: " + string(target);
    }
}
